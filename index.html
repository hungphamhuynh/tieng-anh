let currentQuestionIndex = 0; function shuffleArray(array) { return array.sort(() => Math.random() - 0.5); } function showQuestion() { const q = questions[currentQuestionIndex]; document.getElementById('question').innerHTML = ` Câu ${q.number} (${q.point} Điểm)
${q.question} `; const shuffledOptions = shuffleArray([...q.options]); const answersDiv = document.getElementById('answers'); answersDiv.innerHTML = ''; shuffledOptions.forEach(option => { const btn = document.createElement('button'); btn.textContent = option; btn.className = 'answer'; btn.onclick = () => checkAnswer(option, q.correct, q); answersDiv.appendChild(btn); }); document.getElementById('result').textContent = ''; } function checkAnswer(selected, correct, q) { const buttons = document.querySelectorAll('.answer'); buttons.forEach(btn => { if (btn.textContent === correct) btn.classList.add('correct'); if (btn.textContent === selected && selected !== correct) btn.classList.add('wrong'); btn.disabled = true; }); const result = document.getElementById('result'); if (selected === correct) { result.textContent = `Chính xác! ✅ Đây là ${q.exam}, ${q.point} điểm.`; } else { result.textContent = `Sai ❌. Đáp án đúng là: ${correct}. Đây là ${q.exam}, ${q.point} điểm.`; } } // Hiển thị câu đầu tiên showQuestion(); // Nút câu tiếp theo const nextBtn = document.createElement('button'); nextBtn.textContent = 'Câu tiếp theo'; nextBtn.onclick = () => { currentQuestionIndex = (currentQuestionIndex + 1) % questions.length; showQuestion(); }; document.body.appendChild(nextBtn);
